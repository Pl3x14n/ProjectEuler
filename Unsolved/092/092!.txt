DIM schleife(100000)

FOR i = 1 TO 10000000
   LET i$ = STR$(i)
   LET schleife(1) = i
    
   LET n=1
   DO UNTIL schleife(n) = 1 OR schleife(n) = 89 
    
    
      LET schleife$ = STR$(schleife(n))
       
       
      FOR j = 1 TO LEN(schleife$)
         LET schleife(n+1) = schleife(n+1) + VAL(schleife$(j:j))^2
      NEXT j
       
      LET n = n+1
   LOOP
    
   IF schleife(n) = 89 THEN LET n89 = n89 + 1
NEXT i

PRINT n89
END
