DIM ta(1000000)            !Teileranzahl

FOR n = 2 TO 10^7-1
   LET f = 0            !Teiler von n
    
   FOR a= 1 TO SQR(n)
      IF MOD(n,a)=0 THEN
         LET f=f+1
      END IF 
   NEXT a
    
   LET ta(n) = f*2
   if sqr(n) = int(sqr(n)) then let ta(n) = ta(n) - 1
    
   IF ta(n-1) = ta(n) THEN LET sum = sum + 1
    
    
NEXT n

PRINT "-----------------------------------"
PRINT sum

END
