OPTION ARITHMETIC RATIONAL

FOR d = 2 TO 1000000
   LET proper = proper + phi(d)
NEXT d

PRINT proper


FUNCTION phi(a)
   local b,z,n,r,p
    
   FOR b = 1 TO a-1
    
    
      LET z = a
      LET n = b
      DO 
         LET r = MOD(z,n)
         IF r = 0 THEN EXIT DO 
          
         LET z = n
         LET n = r
      LOOP  
       
      IF n = 1 THEN LET p = p + 1
       
   NEXT b
    
   LET phi = p
    
END FUNCTION 

END
